# 快速上手

通过 <n-f2> 形式调用，设置 ref 即可

<template>
	<view>
		<n-f2 ref="f2"></n-f2>
	</view>
</template>

<script>
	export default {
		data() {
			return {
				source: [
					{name: '芳华',percent: 0.4,a: '1'},
					{name: '妖猫传',percent: 0.2,a: '1'},
					{name: '机器之血',percent: 0.18,a: '1'},
					{name: '心理罪',percent: 0.15,a: '1'},
					{name: '寻梦环游记',percent: 0.05,a: '1'},
					{name: '其他',percent: 0.02,a: '1'}
				]
			}
		},
		onReady() {
			this.f2Chart(this.$refs['f2'], this.source);
		},
		methods: {
			f2Chart(ref, source) {
				this.$store.tools.F2(ref, source, config => {
					const F2 = F2 || ref.F2;
					const chart = new F2.Chart(config);
					//=======start：设置图表显示参数=======
					
					chart.source(source);
					chart.legend({
						position: 'right',
						itemFormatter: val => val
					});
					chart.tooltip(false);
					chart.coord('polar', {
						transposed: true,
						radius: 0.85
					});
					chart.axis(false);
					chart
						.interval()
						.position('a*percent')
						.color('name', ['#1890FF', '#13C2C2', '#2FC25B', '#FACC14', '#F04864', '#8543E0'])
						.adjust('stack')
						.style({
							lineWidth: 1,
							stroke: '#fff',
							lineJoin: 'round',
							lineCap: 'round'
						})
						.animate({
							appear: {
								duration: 1200,
								easing: 'bounceOut'
							}
						});
						
					//=======end：渲染图表=======
					chart.render();
					// 需要把 chart 返回
					return chart;
				})
			}
		}
	}
</script>
